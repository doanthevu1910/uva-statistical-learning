# Best cut-off parameter
data (Carseats)
attach (Carseats)
High=ifelse (Sales<=8," No"," Yes ")
Carseats =data.frame(Carseats ,High)
n.grid=100
grid=seq(0.01,0.99, length=n.grid)
mse.rf=rep(NA,n.grid)
for (i in 1:n.grid){
  set.seed(2020)
  RF.fit=randomForest(High~.,cutoff=c(grid[i],1-grid[i]),data=Carseats[train,])
  RF.pred=predict(RF.fit, newdata=Carseats[test,])
  mse.rf[i]=mean( RF.pred!=Carseats[test,]$High )
}
bestcut=grid[which.min(mse.rf)]
print(bestcut)
